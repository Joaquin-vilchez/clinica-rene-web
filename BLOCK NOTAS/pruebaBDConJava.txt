DROP DATABASE sistemaRene;
CREATE DATABASE sistemaRene;
USE sistemaRene;

CREATE TABLE rolesClinica(
    id_Rol INT PRIMARY KEY AUTO_INCREMENT,
    nombre_Rol VARCHAR(30) NOT NULL UNIQUE,
    descripcion_Rol VARCHAR(100),
    tipoAcceso_Rol ENUM('Escritorio','Web') NOT NULL,
    estado_Rol ENUM('Activo','Inactivo') DEFAULT 'Activo',
    fCrear_Rol TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fActualizacion_Rol TIMESTAMP NULL
);

CREATE TABLE personas (
    id_Persona INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    apellido_paterno VARCHAR(100) NOT NULL,
    apellido_materno VARCHAR(100),
    telefono VARCHAR(20),
    nacionalidad VARCHAR(30),
    numero_documento VARCHAR(15) NOT NULL UNIQUE,
    genero ENUM('M','F','Otro') NOT NULL,
    fecha_nacimiento DATE,
    direccion VARCHAR(200)
);

CREATE TABLE especialidades (
    id_Especialidad INT AUTO_INCREMENT PRIMARY KEY,
    nombre_Especialidad VARCHAR(100) NOT NULL UNIQUE,
    descripcion_Especialidad TEXT,
    detalles_especialidad varchar(50)
);

CREATE TABLE consultorios (
    id_Consultorio INT AUTO_INCREMENT PRIMARY KEY,
    id_Especialidad INT NOT NULL,
    numero_Consultorio VARCHAR(10) NOT NULL UNIQUE,
    piso INT NOT NULL,
    estado_Consultorio ENUM('Disponible', 'Ocupado', 'Mantenimiento') DEFAULT 'Disponible',

    FOREIGN KEY (id_Especialidad) REFERENCES especialidades(id_Especialidad)
);

CREATE TABLE usuariosClinica(
    id_Users INT AUTO_INCREMENT PRIMARY KEY,
    id_Rol INT NOT NULL,
    id_Persona INT NOT NULL UNIQUE,
    email_Users VARCHAR(100) UNIQUE,
    pass_Users VARCHAR(255) NOT NULL,
    fCreacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fUltima_Conex_Users TIMESTAMP NULL,
    tokenRecup_Users VARCHAR(255) NULL,
    fToken_Users TIMESTAMP NULL,
    estado_Users ENUM('Activo','Suspendido','Bloqueado') DEFAULT 'Activo',

    FOREIGN KEY (id_Rol) REFERENCES rolesClinica(id_Rol),
    FOREIGN KEY (id_Persona) REFERENCES personas(id_Persona)
);

CREATE TABLE doctores(
    id_Doctor INT AUTO_INCREMENT PRIMARY KEY,
    id_Persona INT NOT NULL,
    id_Especialidad INT NOT NULL,
    numero_colegiatura VARCHAR(50) UNIQUE NOT NULL,
	

    FOREIGN KEY (id_Persona) REFERENCES personas(id_Persona),
    FOREIGN KEY (id_Especialidad) REFERENCES especialidades(id_Especialidad)
);

CREATE TABLE horarios_doctores (
    id_Horario INT AUTO_INCREMENT PRIMARY KEY,
    id_Doctor INT NOT NULL,
    id_Consultorio INT NOT NULL,
    dia_semana ENUM('Lunes','Martes','Miercoles','Jueves','Viernes','Sabado','Domingo') NOT NULL,
    hora_inicio TIME NOT NULL,
    hora_fin TIME NOT NULL,
    estado_Horario ENUM('Activo','Inactivo') DEFAULT 'Activo',

    FOREIGN KEY (id_Doctor) REFERENCES doctores(id_Doctor),
    FOREIGN KEY (id_Consultorio) REFERENCES consultorios(id_Consultorio),

    UNIQUE KEY unique_doctor_horario (id_Doctor, dia_semana, hora_inicio)
);

CREATE TABLE admision (
    id_Admision INT AUTO_INCREMENT PRIMARY KEY,
    id_Persona INT NOT NULL,
    turno ENUM('Manana', 'Tarde'),

    FOREIGN KEY (id_Persona) REFERENCES personas(id_Persona)
);

CREATE TABLE enfermeras(
    id_Enfermera INT AUTO_INCREMENT PRIMARY KEY,
    id_Persona INT NOT NULL,
    turno ENUM('Manana','Tarde'),

    FOREIGN KEY (id_Persona) REFERENCES personas(id_Persona)
);

CREATE TABLE pacientes(
    id_Paciente INT AUTO_INCREMENT PRIMARY KEY,
    id_Persona INT NOT NULL,
    tipo_sangre ENUM('A+','A-','B+','B-','AB+','AB-','O+','O-'),
    alergias TEXT,
    antecedentes_medicos TEXT,
    id_Tutor_Persona INT NULL,

    FOREIGN KEY (id_Persona) REFERENCES personas(id_Persona),
    FOREIGN KEY (id_Tutor_Persona) REFERENCES personas(id_Persona)
);

CREATE TABLE citas (
    id_Cita INT AUTO_INCREMENT PRIMARY KEY,
    id_Paciente INT NOT NULL,
    id_Doctor INT NOT NULL,
    id_Especialidad INT NOT NULL,
    id_Consultorio INT NOT NULL,

    fechaHora_Cita DATETIME NOT NULL,
    duracion_minutos INT DEFAULT 30,

    estado_Cita ENUM('Programada', 'Confirmada', 'En Espera', 'Completada', 'Cancelada', 'No Asistio') NOT NULL DEFAULT 'Programada',
    estado_triaje ENUM('Pendiente', 'En Proceso', 'Completado') DEFAULT 'Pendiente',
    id_Triaje INT NULL,
    tipo_consulta ENUM('consulta general', 'seguimiento', 'control') DEFAULT 'consulta general',
    motivo_Consulta TEXT,
    fCreacion_Cita TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_Paciente) REFERENCES pacientes(id_Paciente),
    FOREIGN KEY (id_Doctor) REFERENCES doctores(id_Doctor),
    FOREIGN KEY (id_Especialidad) REFERENCES especialidades(id_Especialidad),
    FOREIGN KEY (id_Consultorio) REFERENCES consultorios(id_Consultorio),

    UNIQUE KEY unique_doctor_horario (id_Doctor, fechaHora_Cita),
    UNIQUE KEY unique_consultorio_horario (id_Consultorio, fechaHora_Cita)
);

-- Tabla de pagos (versión correcta con foreign key)
CREATE TABLE pagos (
    id_Pago INT PRIMARY KEY AUTO_INCREMENT,
    id_Cita INT NOT NULL,
    id_Paciente INT NOT NULL,
    monto DECIMAL(10,2) NOT NULL,
    metodo_Pago ENUM('Efectivo', 'Tarjeta Debito', 'Transferencia') DEFAULT 'Efectivo',
    estado_Pago ENUM('Pendiente', 'Pagado', 'Vencido', 'Anulado') DEFAULT 'Pendiente',
    fecha_Limite DATETIME NOT NULL COMMENT 'Fecha limite para pagar (24h antes)',
    fecha_Pago DATETIME COMMENT 'Cuando realmente se pago',
    fCreacion_Pago DATETIME DEFAULT CURRENT_TIMESTAMP,
    numero_Comprobante VARCHAR(50) UNIQUE,
    tipo_Comprobante ENUM('Boleta', 'Factura') DEFAULT 'Boleta',
    numero_Operacion VARCHAR(50) COMMENT 'Para tarjeta/transferencia',
    observaciones TEXT,
    id_Usuario_Registro INT COMMENT 'Usuario que registro el pago',
    FOREIGN KEY (id_Cita) REFERENCES citas(id_Cita),
    FOREIGN KEY (id_Paciente) REFERENCES pacientes(id_Paciente),
    FOREIGN KEY (id_Usuario_Registro) REFERENCES usuariosClinica(id_Users),
    
    INDEX idx_cita (id_Cita),
    INDEX idx_paciente (id_Paciente),
    INDEX idx_estado (estado_Pago),
    INDEX idx_fecha_limite (fecha_Limite)
) COMMENT 'Gestion de pagos de consultas medicas';


CREATE TABLE triajes (
    id_Triaje INT AUTO_INCREMENT PRIMARY KEY,
    id_Cita INT NOT NULL,
    id_Enfermera INT NOT NULL,
    presion_arterial VARCHAR(20),
    frecuencia_cardiaca INT,
    frecuencia_respiratoria INT,
    temperatura DECIMAL(4,2),
    saturacion_oxigeno INT,
    peso DECIMAL(5,2),
    talla DECIMAL(6,2),
    imc DECIMAL(4,2),
    tipo_sangre VARCHAR(5),
    glucemia INT,
    alergias_actualizadas TEXT,
    motivo_consulta_detallado TEXT,
    sintomas_principales TEXT,
    nivel_urgencia ENUM('Baja', 'Media', 'Alta') DEFAULT 'Baja',
    observaciones TEXT,
    fecha_triaje TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado_Triaje ENUM('Pendiente', 'Completado') DEFAULT 'Pendiente',

    FOREIGN KEY (id_Cita) REFERENCES citas(id_Cita),
    FOREIGN KEY (id_Enfermera) REFERENCES enfermeras(id_Enfermera)
);

#CONSULTAS
CREATE TABLE consultas (
    id_Consulta INT AUTO_INCREMENT PRIMARY KEY,
    id_Cita INT NOT NULL,
    id_Doctor INT NOT NULL,
    id_Paciente INT NOT NULL,

    diagnostico TEXT NOT NULL,
    sintomas TEXT,
    observaciones TEXT,
    tratamiento TEXT,
    fecha_consulta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_Cita) REFERENCES citas(id_Cita),
    FOREIGN KEY (id_Doctor) REFERENCES doctores(id_Doctor),
    FOREIGN KEY (id_Paciente) REFERENCES pacientes(id_Paciente),

    INDEX idx_doctor (id_Doctor),
    INDEX idx_paciente (id_Paciente),
    INDEX idx_fecha (fecha_consulta)
);

CREATE TABLE estados_cita (
    id_estado INT AUTO_INCREMENT PRIMARY KEY,
    descripcion VARCHAR(50) NOT NULL
);

#CATALOGO
CREATE TABLE medicamentos (
    id_Medicamento INT AUTO_INCREMENT PRIMARY KEY,
    nombre_comercial VARCHAR(100) NOT NULL,
    nombre_generico VARCHAR(100),
    presentacion VARCHAR(50),
    concentracion VARCHAR(50),
    via_administracion ENUM('Oral','Intravenosa','Intramuscular','Topica','Sublingual','Inhalatoria','Rectal','Otra') NOT NULL,
    tipo_medicamento ENUM('Analgesico','Antibiotico','Antiinflamatorio','Antihipertensivo','Antidiabetico','Vitamina','Otro') DEFAULT 'Otro',
    estado ENUM('Activo','Inactivo') DEFAULT 'Activo',

    INDEX idx_nombre_comercial (nombre_comercial),
    INDEX idx_tipo (tipo_medicamento)
);

#RECETAS
CREATE TABLE recetas (
    id_Receta INT AUTO_INCREMENT PRIMARY KEY,
    id_Consulta INT NOT NULL,
    id_Doctor INT NOT NULL,
    id_Paciente INT NOT NULL,

    fecha_emision TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    observaciones TEXT,
    estado ENUM('Activa','Vencida','Anulada') DEFAULT 'Activa',

    FOREIGN KEY (id_Consulta) REFERENCES consultas(id_Consulta),
    FOREIGN KEY (id_Doctor) REFERENCES doctores(id_Doctor),
    FOREIGN KEY (id_Paciente) REFERENCES pacientes(id_Paciente),

    INDEX idx_doctor (id_Doctor),
    INDEX idx_paciente (id_Paciente),
    INDEX idx_fecha (fecha_emision)
);

#DETALLE DE RECETAS
CREATE TABLE detalle_recetas (
    id_Detalle INT AUTO_INCREMENT PRIMARY KEY,
    id_Receta INT NOT NULL,
    id_Medicamento INT NOT NULL,

    dosis VARCHAR(100) NOT NULL,
    frecuencia VARCHAR(100) NOT NULL,
    duracion VARCHAR(50) NOT NULL,
    cantidad INT NOT NULL,
    indicaciones TEXT,

    FOREIGN KEY (id_Receta) REFERENCES recetas(id_Receta) ON DELETE CASCADE,
    FOREIGN KEY (id_Medicamento) REFERENCES medicamentos(id_Medicamento)
);

#ORDENES
CREATE TABLE ordenes_laboratorio (
    id_Orden INT AUTO_INCREMENT PRIMARY KEY,
    id_Consulta INT NOT NULL,
    id_Doctor INT NOT NULL,
    id_Paciente INT NOT NULL,

    tipo_examen VARCHAR(100) NOT NULL,
    indicaciones TEXT,
    estado ENUM('Pendiente','En Proceso','Completado','Cancelado') DEFAULT 'Pendiente',

    fecha_orden TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_resultado TIMESTAMP NULL,
    resultado TEXT,
    observaciones_resultado TEXT,

    FOREIGN KEY (id_Consulta) REFERENCES consultas(id_Consulta),
    FOREIGN KEY (id_Doctor) REFERENCES doctores(id_Doctor),
    FOREIGN KEY (id_Paciente) REFERENCES pacientes(id_Paciente),

    INDEX idx_doctor (id_Doctor),
    INDEX idx_paciente (id_Paciente),
    INDEX idx_estado (estado)
);
